-- Enhanced Ethereum Price Analysis with Volume, Volatility, and Market Insights
WITH detailed_prices AS (
    SELECT
        minute AS time,
        price
    FROM prices.usd
    WHERE symbol = 'ETH'
        AND contract_address IS NULL
        AND minute >= TIMESTAMP '2024-01-01 00:00:00'
        AND minute < TIMESTAMP '2026-01-01 00:00:00'
),
daily_metrics AS (
    SELECT
        date_trunc('day', time) AS day,
        AVG(price) AS daily_price,
        STDDEV(price) AS daily_stddev,
        MAX(price) AS daily_high,
        MIN(price) AS daily_low,
        COUNT(*) AS price_points
    FROM detailed_prices
    GROUP BY 1
),
price_changes AS (
    SELECT
        day,
        daily_price,
        daily_high,
        daily_low,
        daily_stddev,
        LAG(daily_price) OVER (ORDER BY day) AS previous_price,
        (daily_price - LAG(daily_price) OVER (ORDER BY day)) AS price_change,
        (daily_price - LAG(daily_price) OVER (ORDER BY day)) / LAG(daily_price) OVER (ORDER BY day) * 100 AS pct_change
    FROM daily_metrics
),
volatility_bands AS (
    SELECT
        day,
        daily_price,
        AVG(daily_price) OVER (ORDER BY day ROWS BETWEEN 29 PRECEDING AND CURRENT ROW) AS ma_30,
        STDDEV(daily_price) OVER (ORDER BY day ROWS BETWEEN 29 PRECEDING AND CURRENT ROW) AS volatility_30d
    FROM daily_metrics
)
SELECT
    d.time,
    d.price AS precise_price,
    m.daily_high,
    m.daily_low,
    m.daily_stddev,
    pc.price_change,
    pc.pct_change,
    vb.ma_30 AS moving_avg_30d,
    vb.volatility_30d,
    CASE
        WHEN pc.price_change > 0 THEN pc.price_change
        ELSE NULL
    END AS positive_price_change,
    CASE
        WHEN pc.price_change < 0 THEN pc.price_change
        ELSE NULL
    END AS negative_price_change,
    -- Support/Resistance Indicators
    MIN(m.daily_low) OVER (ORDER BY m.day ROWS BETWEEN 29 PRECEDING AND CURRENT ROW) AS support_30d,
    MAX(m.daily_high) OVER (ORDER BY m.day ROWS BETWEEN 29 PRECEDING AND CURRENT ROW) AS resistance_30d
FROM detailed_prices d
    LEFT JOIN daily_metrics m ON date_trunc('day', d.time) = m.day
    LEFT JOIN price_changes pc ON date_trunc('day', d.time) = pc.day
    LEFT JOIN volatility_bands vb ON date_trunc('day', d.time) = vb.day
ORDER BY d.time;
